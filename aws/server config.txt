sudo nano /etc/systemd/system/gunicorn.service


[Unit]
Description=gunicorn daemon
Requires=gunicorn.socket
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/rfinv/rfinv
ExecStart=/home/ubuntu/rfinv/isenv/bin/gunicorn \
          --access-logfile - \
          --workers 3 \
          --bind unix:/run/gunicorn.sock \
          rfinv.wsgi:application

[Install]
WantedBy=multi-user.target
